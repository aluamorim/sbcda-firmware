70 /******************************************************************************
71  * NOTE 1: This project provides two demo applications. A simple blinky style
72  * project, and a more comprehensive test and demo application. The
73  * configCREATE_SIMPLE_BLINKY_DEMO_ONLY setting in FreeRTOSConifg.h is used to
74  * select between the two. See the notes on using
75  * configCREATE_SIMPLE_BLINKY_DEMO_ONLY in main.c. This file implements the
76  * simply blinky style version.
77  *
78  * NOTE 2: This file only contains the source code that is specific to the
79  * basic demo. Generic functions, such FreeRTOS hook functions, and functions
80  * required to configure the hardware, are defined in main.c.
81  ******************************************************************************
82  *
83  * main_blinky() creates one queue, and two tasks. It then starts the
84  * scheduler.
85  *
86  * The Queue Send Task:
87  * The queue send task is implemented by the prvQueueSendTask() function in
88  * this file. prvQueueSendTask() sits in a loop that causes it to repeatedly
89  * block for 200 milliseconds, before sending the value 100 to the queue that
90  * was created within main_blinky(). Once the value is sent, the task loops
91  * back around to block for another 200 milliseconds.
92  *
93  * The Queue Receive Task:
94  * The queue receive task is implemented by the prvQueueReceiveTask() function
95  * in this file. prvQueueReceiveTask() sits in a loop where it repeatedly
96  * blocks on attempts to read data from the queue that was created within
97  * main_blinky(). When data is received, the task checks the value of the
98  * data, and if the value equals the expected 100, toggles the LED. The 'block
99  * time' parameter passed to the queue receive function specifies that the
100  * task should be held in the Blocked state indefinitely to wait for data to
101  * be available on the queue. The queue receive task will only leave the
102  * Blocked state when the queue send task writes to the queue. As the queue
103  * send task writes to the queue every 200 milliseconds, the queue receive
104  * task leaves the Blocked state every 200 milliseconds, and therefore toggles
105  * the LED every 200 milliseconds.
106  */

159 void blinky_rtos_app(void) {
160 
161  /*
162  * Initialize MSS GPIOs.
163  */
164  MSS_GPIO_init();
165 
166  /*
167  * Configure MSS GPIOs.
168  */
169  MSS_GPIO_config(DS3_LED_GPIO, MSS_GPIO_OUTPUT_MODE);
170  MSS_GPIO_config(DS4_LED_GPIO, MSS_GPIO_OUTPUT_MODE);
171 
172  /*
173  * Set LEDs to initial state
174  */
175  MSS_GPIO_set_outputs(MSS_GPIO_get_outputs() | DS3_LED_MASK | DS4_LED_MASK);
176 
177  /* Create the queue. */
178  xQueue = xQueueCreate( mainQUEUE_LENGTH, sizeof(unsigned long));
179 
180  if (xQueue != NULL) {
181  /* Start the two tasks as described in the comments at the top of this
182  file. */
183  xTaskCreate(prvQueueReceiveTask, /* The function that implements the task. */
184  "Rx", /* The text name assigned to the task - for debug only as it is not used by the kernel. */
185  configMINIMAL_STACK_SIZE, /* The size of the stack to allocate to the task. */
186  (void *) mainQUEUE_RECEIVE_PARAMETER, /* The parameter passed to the task - just to check the functionality. */
187  mainQUEUE_RECEIVE_TASK_PRIORITY, /* The priority assigned to the task. */
188  NULL); /* The task handle is not required, so NULL is passed. */
189 
190  xTaskCreate(prvQueueSendTask, "TX", configMINIMAL_STACK_SIZE,
191  (void *) mainQUEUE_SEND_PARAMETER, mainQUEUE_SEND_TASK_PRIORITY,
192  NULL);
193 
194  /* Start the tasks and timer running. */
195  vTaskStartScheduler();
196  }
197 
198  /* If all is well, the scheduler will now be running, and the following
199  line will never be reached. If the following line does execute, then
200  there was insufficient FreeRTOS heap memory available for the idle and/or
201  timer tasks to be created. See the memory management section on the
202  FreeRTOS web site for more details. */
203  for (;;)
204  ;
205 }

208 static void prvQueueSendTask(void *pvParameters) {
209  TickType_t xNextWakeTime;
210  const unsigned long ulValueToSend = 100UL;
211 
212  /* Check the task parameter is as expected. */
213  configASSERT( ( ( unsigned long ) pvParameters ) == mainQUEUE_SEND_PARAMETER );
214 
215  /* Initialise xNextWakeTime - this only needs to be done once. */
216  xNextWakeTime = xTaskGetTickCount();
217 
218  for (;;) {
219  /* Place this task in the blocked state until it is time to run again.
220  The block time is specified in ticks, the constant used converts ticks
221  to ms. While in the Blocked state this task will not consume any CPU
222  time. */
223  vTaskDelayUntil(&xNextWakeTime, mainQUEUE_SEND_FREQUENCY_MS);
224 
225  /* Send to the queue - causing the queue receive task to unblock and
226  toggle the LED. 0 is used as the block time so the sending operation
227  will not block - it shouldn't need to block as the queue should always
228  be empty at this point in the code. */
229  xQueueSend(xQueue, &ulValueToSend, 0U);
230  }
231 }

234 static void prvQueueReceiveTask(void *pvParameters) {
235  unsigned long ulReceivedValue;
236 
237  /* Check the task parameter is as expected. */
238  configASSERT( ( ( unsigned long ) pvParameters ) == mainQUEUE_RECEIVE_PARAMETER );
239  int cnt = 4;
240  for (;;) {
241  /* Wait until something arrives in the queue - this task will block
242  indefinitely provided INCLUDE_vTaskSuspend is set to 1 in
243  FreeRTOSConfig.h. */
244  xQueueReceive(xQueue, &ulReceivedValue, portMAX_DELAY);
245 
246  /* To get here something must have been received from the queue, but
247  is it the expected value? If it is, toggle the LED. */
248  if (ulReceivedValue == 100UL) {
249  cnt = ~cnt;
250  vParTestToggleLED(cnt);
251  ulReceivedValue = 0U;
252  }
253  }
254 }
