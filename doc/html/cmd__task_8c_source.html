23 void Cmd_RTC_Set(Command_Typedef * cmd) {
24 
25  mss_rtc_calendar_t cal;
26 
27  if (!RTC_Is_Initialized()) {
28  RTC_Setup();
29  }
30  /*
31  * RTC 'month', 'day', 'week' and 'weekday' parameters
32  * require a minimum value of 1. If zero (or less) is
33  * provided the RTC drives halts the processor.
34  * That is stupid but is the way the driver
35  * was developed.
36  */
37  cal.year = cmd->param[0]; // see command specification table
38  cal.month = cmd->param[1]>0?cmd->param[1]:1;
39  cal.day = cmd->param[2]>0?cmd->param[2]:1;
40  cal.hour = cmd->param[3];
41  cal.minute = cmd->param[4];
42  cal.second = cmd->param[5];
43  cal.week = cmd->param[6]>0?cmd->param[6]:1;
44  cal.weekday = cmd->param[7]>0?cmd->param[7]:1;
45  RTC_Set(&cal);
46  RTC_Start();
47 }

70 
71  uint8_t rx_buffer[COMM_RX_BUFFER_SIZE];
72  uint8_t tx_buffer[COMM_TX_BUFFER_SIZE];
73  uint8_t *buffer_ptr;
74  tx_buffer[0] = 0x10;
75 
76 
77  uint8_t cmd_buffer[CMD_MAX_SIZE_IN_BYTES];
78  Command_Typedef * cmd_received;
79  CommandResult_Typedef cmd_result = CMD_OK;
80  CommandMode_Typedef cmd_mode = CMD_PTT_MODE;
81  uint32_t cmd_data;
82 
83  Comm_Setup(tx_buffer, rx_buffer); // initializes comm layer
84 
85  buffer_ptr = (uint8_t*) PTT_GetReadyFifoOutput();
86  Comm_Set_Tx_Buffer(buffer_ptr, PTT_TOTAL_RECORD_SIZE);
87 
88 #ifdef DEBUG_PRINT
89  print("\n\r Command_Parser_Task started... \n\r");
90 #endif
91  for (;;) {
92 
93  // waits until next command is received
94  while (Comm_Get_Bytes_Available() == 0);
95  cmd_received = (Command_Typedef *) Comm_Get_Rx_Buffer();
96 
97  switch (cmd_received->id) {
98  case CMD_SET_CONFIG_MODE:
99  /*
100  * The configuration mode is used to setup configuration
101  * patterns to the Decoder_Unit hardware. This command
102  * must be sent before any other configuration command
103  * (RTC_SET, HK_CONFIG, etc.) in order
104  * to allow the OS to suspend the PTT package and respond
105  * all other requests in time.
106  */
107  //PTT_Service_Pause();
108  cmd_mode = CMD_CONFIG_MODE;
109  cmd_result = CMD_OK;
110  cmd_data = 0;
111  Comm_Set_Tx_Buffer(tx_buffer, COMM_TX_BUFFER_SIZE);
112  break;
113  case CMD_SET_HK_MODE:
114  /*
115  * Setting the parser operation to HK_MODE will simply
116  * change the current output buffer to the HK_OUTPUT_BUFFER.
117  * There is no ack in this mode, the master simply reads the
118  * current HK package and verifies its type (first byte of
119  * every package) and data. These are the only ways for the
120  * master to check if this command was properly executed
121  */
122  Comm_Set_Tx_Buffer(HK_Get_Buffer(), HOUSEKEEPING_PACKAGE_SIZE);
123  cmd_mode = CMD_HK_MODE;
124  cmd_result = CMD_OK;
125  cmd_data = 0;
126  break;
127  case CMD_SET_PTT_MODE:
128  /*
129  * This command sets the current ouput buffer ptr to the
130  * default READY_FIFO_OUTPUT for ptt packages. There is no
131  * ACK repply to this command. In order to check if it was
132  * executed properly, the master has to check whether
133  * the received packages are of PTT_TYPE (first byte).
134  */
135  //PTT_Service_Resume();
136  Comm_Set_Tx_Buffer(PTT_GetReadyFifoOutput(), PTT_TOTAL_RECORD_SIZE);
137  cmd_mode = CMD_PTT_MODE;
138  cmd_result = CMD_OK;
139  cmd_data = 0;
140  break;
142  case CMD_SET_ADC_MODE:
143  /*
144  * This command sets the current ouput buffer ptr to the
145  * default ADC_BUFFER_OUTPUT for adc sample packages.
146  */
147  //PTT_Service_Pause();
148  Comm_Set_Tx_Buffer((uint8_t*) ADC_GetSamplerBuffer(), ADC_BUFFER_SIZE_IN_BYTES);
149  cmd_mode = CMD_ADC_MODE;
150  cmd_result = CMD_OK;
151  cmd_data = 0;
152  break;
154  case ECHO:
155  print("\n\r> Echo: ");
156  print_hex((uint8_t*) cmd_received, 10);
157  print("\n\r");
158  cmd_result = CMD_OK;
159  cmd_data = 0;
160  break;
161  case RTC_SET:
162  Cmd_RTC_Set(cmd_received);
163  cmd_result = CMD_OK;
164  cmd_data = 0;
165  break;
166  case RTC_PAUSE:
167  RTC_Stop();
168  cmd_result = CMD_OK;
169  cmd_data = 0;
170  break;
171  case RTC_RESUME:
172  RTC_Start();
173  cmd_result = CMD_OK;
174  cmd_data = 0;
175  break;
176  case RTC_RESET:
177  RTC_Reset();
178  cmd_result = CMD_OK;
179  cmd_data = 0;
180  break;
181  case HK_CONFIG:
182  Cmd_HK_Set_Config(cmd_received);
183  cmd_result = CMD_OK;
184  cmd_data = 0;
185  break;
186  case PTT_AVAILABLE:
187  /*
188  * PTT_AVAILABLE must be executed in CMD_MODE
189  * in order to generate a valid ACK response
190  */
191  cmd_data = PTT_ReadyPackages_Available();
192  cmd_result = CMD_OK;
193  break;
194 // case PTT_POP: // executed in the Comm_Write_Handler ISR
195 // /*
196 // * PTT_POP doesn't require an ACK response.
197 // * The master simply sends the PTT_POP and then
198 // * checks if the read PTT_PACKAGE was updated.
199 // * PTT_POP can be executed in both PTT and CMD
200 // * modes.
201 // */
202 // PTT_SetPackageReadMutex();
203 // while (PTT_GetPackageReadMutex())
204 // ;
205 // cmd_result = CMD_OK;
206 // cmd_data = 0;
207 // break;
208  case PTT_PAUSE:
209  PTT_Service_Pause();
210  cmd_result = CMD_OK;
211  cmd_data = 0;
212  break;
213  case PTT_RESUME:
214  PTT_Service_Resume();
215  cmd_result = CMD_OK;
216  cmd_data = 0;
217  break;
218  case PTT_ISPAUSED:
219  /*
220  * PTT_ISPAUSED must be executed in CMD_MODE
221  * in order to generate a valid ACK response
222  */
223  cmd_data = PTT_Service_isPaused() & 0x01;
224  cmd_result = CMD_OK;
225  cmd_data = 0;
226  break;
227  case ADC_LOAD:
228  ADC_SetState(ADC_BUSY);
229  cmd_result = CMD_OK;
230  cmd_data = 0;
231  break;
232  case ADC_STATE:
233  cmd_result = CMD_OK;
234  cmd_data = ADC_GetState();
235  break;
236  case ADC_RESET:
237  ADC_SetState(ADC_IDLE);
238  cmd_result = CMD_OK;
239  cmd_data = ADC_GetState();
240  break;
241  default:
242  cmd_result = CMD_FAILED;
243  cmd_data = 0;
244  break;
245  }
246  if (cmd_mode == CMD_CONFIG_MODE) { // in config mode all commands have ack
247  tx_buffer[0] = ACK_PCKG_ID;
248  tx_buffer[1] = cmd_received->id;
249  tx_buffer[2] = cmd_result;
250  // some commands may send back a 32 bit data value as result (MSB)
251  tx_buffer[3] = cmd_data >> 24;
252  tx_buffer[4] = cmd_data >> 16;
253  tx_buffer[5] = cmd_data >> 8;
254  tx_buffer[6] = cmd_data;
255  tx_buffer[7] = 0;
256  tx_buffer[8] = 0;
257  Comm_Set_Tx_Buffer(tx_buffer, COMM_TX_BUFFER_SIZE);
258  }
259  Comm_Set_Bytes_Available(0);
260  Comm_Set_Slave_Mode();
261 
262  }
