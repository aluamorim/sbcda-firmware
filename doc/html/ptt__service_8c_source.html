209 uint8_t PTT_PushFreePackage(PTTPackage_Typedef * ptr) {
210 
211  while (__getStackMutex() == 1)
212  ;
213  __setStackMutex();
214  if (freeStackTop >= PTT_PACKAGE_BUFFER_SIZE) {
215  __clearStackMutex();
216  return 0; // free stack is already full
217  } else if (ptr != 0) {
218  freePackagesStack[freeStackTop++] = ptr;
219  }
220  __clearStackMutex();
221 
222  return 1;
